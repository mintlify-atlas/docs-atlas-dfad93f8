This page documents all TypeScript types and interfaces exported by Theme Gen. Types are organized into categories for easy reference.

## Theme Types

### Theme

The core theme type that defines the structure of a complete theme.

```typescript
type Theme = {
  name: string;
  colors: {
    // Core colors
    text: string;
    background: string;
    primary: string;
    container: string;
    accent: string;

    // State colors
    success: string;
    error: string;
    warning: string;

    // On-surface colors (auto-derived)
    onPrimary: string;
    onContainer: string;
    onAccent: string;
    onSuccess: string;
    onError: string;
    onWarning: string;

    // Utility colors (auto-derived)
    border: string;
    muted: string;
    ring: string;
  };
};
```

<ResponseField name="name" type="string" required>
  Display name of the theme (e.g., "Light", "Dark")
</ResponseField>

<ResponseField name="colors" type="object" required>
  Collection of all color values in the theme
  
  <Expandable title="Core Colors">
    <ResponseField name="text" type="string" required>
      Primary text color
    </ResponseField>
    
    <ResponseField name="background" type="string" required>
      Primary background color
    </ResponseField>
    
    <ResponseField name="primary" type="string" required>
      Primary brand color for buttons and key actions
    </ResponseField>
    
    <ResponseField name="container" type="string" required>
      Container background color (cards, panels)
    </ResponseField>
    
    <ResponseField name="accent" type="string" required>
      Accent color for highlights and secondary actions
    </ResponseField>
  </Expandable>
  
  <Expandable title="State Colors">
    <ResponseField name="success" type="string" required>
      Success state color (green)
    </ResponseField>
    
    <ResponseField name="error" type="string" required>
      Error state color (red)
    </ResponseField>
    
    <ResponseField name="warning" type="string" required>
      Warning state color (yellow/orange)
    </ResponseField>
  </Expandable>
  
  <Expandable title="On-Surface Colors">
    <ResponseField name="onPrimary" type="string" required>
      Text color to use on primary background
    </ResponseField>
    
    <ResponseField name="onContainer" type="string" required>
      Text color to use on container background
    </ResponseField>
    
    <ResponseField name="onAccent" type="string" required>
      Text color to use on accent background
    </ResponseField>
    
    <ResponseField name="onSuccess" type="string" required>
      Text color to use on success background
    </ResponseField>
    
    <ResponseField name="onError" type="string" required>
      Text color to use on error background
    </ResponseField>
    
    <ResponseField name="onWarning" type="string" required>
      Text color to use on warning background
    </ResponseField>
  </Expandable>
  
  <Expandable title="Utility Colors">
    <ResponseField name="border" type="string" required>
      Border color derived from text and background
    </ResponseField>
    
    <ResponseField name="muted" type="string" required>
      Muted text color for secondary content
    </ResponseField>
    
    <ResponseField name="ring" type="string" required>
      Focus ring color (typically matches primary)
    </ResponseField>
  </Expandable>
</ResponseField>

### SavedTheme

Represents a user-saved theme with metadata.

```typescript
interface SavedTheme {
  id: string;
  name: string;
  theme: Theme;
  themeName: string;
  savedAt: string;
}
```

<ResponseField name="id" type="string" required>
  Unique identifier for the saved theme (e.g., "saved-1234567890")
</ResponseField>

<ResponseField name="name" type="string" required>
  User-provided name for the saved theme
</ResponseField>

<ResponseField name="theme" type="Theme" required>
  The complete theme object
</ResponseField>

<ResponseField name="themeName" type="string" required>
  Base theme name ("light" or "dark")
</ResponseField>

<ResponseField name="savedAt" type="string" required>
  ISO 8601 timestamp when the theme was saved
</ResponseField>

### ThemeContextType

The type definition for the Theme Context API.

```typescript
type ThemeContextType = {
  theme: Theme;
  themeName: string;
  setTheme: (themeName: string, customColors?: Partial<Theme["colors"]>) => void;
  updateThemeProperty: (path: string[], value: string) => void;
  undo: () => void;
  redo: () => void;
  canUndo: boolean;
  canRedo: boolean;
  pushHistory: () => void;
  savedThemes: SavedTheme[];
  saveCurrentTheme: (name: string) => void;
  deleteSavedTheme: (id: string) => void;
  loadSavedTheme: (saved: SavedTheme) => void;
};
```

<ResponseField name="theme" type="Theme" required>
  Current active theme
</ResponseField>

<ResponseField name="themeName" type="string" required>
  Current theme name ("light" or "dark")
</ResponseField>

<ResponseField name="setTheme" type="function" required>
  Set the theme by name with optional custom colors
  
  **Parameters:**
  - `themeName` (string): Theme to activate
  - `customColors` (Partial&lt;Theme["colors"]&gt;): Optional color overrides
</ResponseField>

<ResponseField name="updateThemeProperty" type="function" required>
  Update a specific theme property by path
  
  **Parameters:**
  - `path` (string[]): Property path (e.g., ["colors", "primary"])
  - `value` (string): New value to set
</ResponseField>

<ResponseField name="undo" type="function" required>
  Undo the last theme change
</ResponseField>

<ResponseField name="redo" type="function" required>
  Redo the last undone change
</ResponseField>

<ResponseField name="canUndo" type="boolean" required>
  Whether undo is available
</ResponseField>

<ResponseField name="canRedo" type="boolean" required>
  Whether redo is available
</ResponseField>

<ResponseField name="pushHistory" type="function" required>
  Manually push current state to history
</ResponseField>

<ResponseField name="savedThemes" type="SavedTheme[]" required>
  Array of user-saved themes
</ResponseField>

<ResponseField name="saveCurrentTheme" type="function" required>
  Save the current theme
  
  **Parameters:**
  - `name` (string): Name for the saved theme
</ResponseField>

<ResponseField name="deleteSavedTheme" type="function" required>
  Delete a saved theme by ID
  
  **Parameters:**
  - `id` (string): ID of theme to delete
</ResponseField>

<ResponseField name="loadSavedTheme" type="function" required>
  Load a saved theme
  
  **Parameters:**
  - `saved` (SavedTheme): Theme to load
</ResponseField>

### WCAGContrastResult

WCAG contrast ratio analysis result.

```typescript
interface WCAGContrastResult {
  ratio: number;
  aaNormal: boolean;
  aaLarge: boolean;
  aaaNormal: boolean;
  aaaLarge: boolean;
}
```

<ResponseField name="ratio" type="number" required>
  Calculated contrast ratio (e.g., 4.5)
</ResponseField>

<ResponseField name="aaNormal" type="boolean" required>
  Passes WCAG AA for normal text (4.5:1)
</ResponseField>

<ResponseField name="aaLarge" type="boolean" required>
  Passes WCAG AA for large text (3:1)
</ResponseField>

<ResponseField name="aaaNormal" type="boolean" required>
  Passes WCAG AAA for normal text (7:1)
</ResponseField>

<ResponseField name="aaaLarge" type="boolean" required>
  Passes WCAG AAA for large text (4.5:1)
</ResponseField>

### HarmonyMode

Color harmony generation mode.

```typescript
type HarmonyMode = "complementary" | "monochromatic";
```

<ResponseField name="complementary" type="string">
  Generate colors 180 degrees apart on the color wheel
</ResponseField>

<ResponseField name="monochromatic" type="string">
  Generate colors using the same hue with different lightness
</ResponseField>
